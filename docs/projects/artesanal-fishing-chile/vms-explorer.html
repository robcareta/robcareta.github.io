This tool explores SERNAPESCA vessel monitoring system (VMS) data for the artisanal fishing fleet in Los Lagos, Chile (2019–2026). The dataset covers ~58.5 million GPS pings. Use the three tabs to explore spatial patterns, distance gradients from aquaculture farms, and temporal trends.

Note

The app runs entirely in your browser via WebAssembly. First load takes ~20–30 seconds while the R environment initialises — please be patient.

# ── Load data (bundled as resources) ─────────────────────────
cell_annual   <- read.csv("data/cell_annual.csv",   stringsAsFactors = FALSE)
ring_annual   <- read.csv("data/ring_annual.csv",   stringsAsFactors = FALSE)
monthly_trend <- read.csv("data/monthly_trend.csv", stringsAsFactors = FALSE)

monthly_trend$yearmonth <- as.Date(monthly_trend$yearmonth)

RING_LEVELS <- c("0\u20132 km","2\u20135 km","5\u201310 km","10\u201320 km","20\u201350 km",">50 km")
ring_annual$ring <- factor(ring_annual$ring, levels = RING_LEVELS)

RING_COLS <- c(
  "0\u20132 km"   = "#b5000a",
  "2\u20135 km"   = "#e05c00",

  "10\u201320 km" = "#4a7c59",
  "20\u201350 km" = "#2a5caa",
  ">50 km"        = "#7a7a7a"
)

YEARS <- sort(unique(cell_annual$year))
CELL  <- 0.05

# ── Server ───────────────────────────────────────────────────
server <- function(input, output, session) {

  # MAP
  map_data <- reactive({
    cell_annual %>% filter(year == input$map_year, n_pings >= input$min_pings)
  })

  output$map <- renderLeaflet({
    leaflet() %>%
      addProviderTiles(providers$CartoDB.Positron) %>%
      setView(lng=-73.5, lat=-42.5, zoom=7)
  })

  observe({
    d <- map_data(); if (nrow(d)==0) return()
    metric <- input$map_metric
    vals   <- d[[metric]]
    pal <- if (metric=="frac")
      colorNumeric("YlOrRd", domain=c(0,1), na.color="transparent")
    else
      colorNumeric("Blues", domain=vals, na.color="transparent")
    half <- CELL/2
    leafletProxy("map", data=d) %>%
      clearShapes() %>% clearControls() %>%
      addRectangles(
        lng1=~lon_bin-half, lat1=~lat_bin-half,
        lng2=~lon_bin+half, lat2=~lat_bin+half,
        fillColor=~pal(d[[metric]]), fillOpacity=0.75,
        color=NA, weight=0,
        popup=~paste0(
          "<b>Lat/Lon:</b> ", round(lat_bin,3),", ",round(lon_bin,3),"<br>",
          "<b>Year:</b> ", year,"<br>",
          "<b>Pings:</b> ", format(n_pings,big.mark=","),"<br>",
          "<b>Fishing:</b> ", round(frac*100,1),"%<br>",
          "<b>Ring:</b> ", ring
        )
      ) %>%
      addLegend("bottomright", pal=pal, values=vals,
        title=if(metric=="frac") "Fishing<br>fraction" else "Log pings",
        labFormat=if(metric=="frac")
          labelFormat(suffix="%",transform=function(x) round(x*100,0))
          else labelFormat())
  })

  output$map_stats <- renderUI({
    d <- map_data(); if (nrow(d)==0) return(NULL)
    tagList(
      tags$div(class="stat-box",
        tags$div(class="stat-val", format(sum(d$n_pings),big.mark=",")),
        tags$div(class="stat-lbl", paste0("Total pings (",input$map_year,")"))
      ),
      tags$div(class="stat-box",
        tags$div(class="stat-val",
          paste0(round(sum(d$n_fishing)/sum(d$n_pings)*100,1),"%")),
        tags$div(class="stat-lbl","Fishing fraction")
      ),
      tags$div(class="stat-box",
        tags$div(class="stat-val", format(nrow(d),big.mark=",")),
        tags$div(class="stat-lbl","Active grid cells")
      )
    )
  })

  # GRADIENT
  output$grad_plot <- renderPlot({
    yrs <- as.integer(input$grad_years)
    if (length(yrs)==0) return(NULL)
    d <- ring_annual %>%
      filter(year %in% yrs, ring != ">50 km") %>%
      mutate(ring=factor(ring, levels=RING_LEVELS))

    if (input$grad_view=="frac") {
      p <- ggplot(d, aes(x=ring, y=weighted_frac*100, fill=ring)) +
        geom_col(position=position_dodge(0.75), width=0.65, alpha=0.9) +
        geom_text(aes(label=paste0(round(weighted_frac*100,1),"%")),
          position=position_dodge(0.75), vjust=-0.4, size=3.2, fontface="bold") +
        scale_fill_manual(values=RING_COLS) +
        scale_y_continuous(limits=c(0,80), labels=label_percent(scale=1,suffix="%")) +
        labs(title="Fishing Fraction by Distance to Nearest Farm",
             x="Distance ring", y="% pings while fishing")
    } else {
      p <- ggplot(d, aes(x=ring, y=total_pings/1e6, fill=ring)) +
        geom_col(position=position_dodge(0.75), width=0.65, alpha=0.9) +
        scale_fill_manual(values=RING_COLS) +
        scale_y_continuous(labels=label_number(suffix="M",accuracy=0.1)) +
        labs(title="Total Pings by Distance Ring",
             x="Distance ring", y="Total pings (millions)")
    }
    if (length(yrs)>1)
      p <- p + facet_wrap(~year, ncol=4) +
        theme_bw(base_size=11) +
        theme(legend.position="none", strip.background=element_rect(fill="grey92"),
              panel.grid.major.x=element_blank(), panel.grid.minor=element_blank(),
              plot.title=element_text(face="bold"))
    else
      p <- p + theme_bw(base_size=12) +
        theme(legend.position="none", panel.grid.major.x=element_blank(),
              panel.grid.minor=element_blank(), plot.title=element_text(face="bold"))
    p
  })

  # TRENDS
  trend_data <- reactive({
    monthly_trend %>%
      filter(year>=input$trend_years[1], year<=input$trend_years[2])
  })

  output$trend_plot <- renderPlot({
    d <- trend_data(); if (nrow(d)==0) return(NULL)
    yvar <- input$trend_metric
    lbl  <- switch(yvar,
      n_pings="Total VMS pings/month", n_fishing="Fishing pings/month",
      frac="Fishing fraction (%)", n_cells="Active grid cells")
    p <- ggplot(d, aes(x=yearmonth, y=.data[[yvar]])) +
      geom_line(color="#005f73", linewidth=0.8) +
      geom_point(color="#005f73", size=1.2, alpha=0.7) +
      geom_smooth(method="loess", span=0.25, se=TRUE,
        color="#ee9b00", fill="#ee9b00", alpha=0.15, linewidth=0.9) +
      scale_x_date(date_breaks="6 months", date_labels="%b %Y") +
      theme_bw(base_size=11) +
      theme(axis.text.x=element_text(angle=35,hjust=1),
            panel.grid.minor=element_blank(),
            plot.title=element_text(face="bold"))
    if (yvar=="frac")
      p <- p + scale_y_continuous(labels=label_percent(scale=100)) +
        labs(title="Monthly Fishing Fraction", x=NULL, y=lbl)
    else if (yvar %in% c("n_pings","n_fishing"))
      p <- p + scale_y_continuous(labels=label_number(scale=1/1e6,suffix="M")) +
        labs(title=paste("Monthly",lbl), x=NULL, y=lbl)
    else
      p <- p + labs(title=paste("Monthly",lbl), x=NULL, y=lbl)
    p
  })

  output$trend_year_plot <- renderPlot({
    d <- trend_data() %>%
      group_by(year) %>%
      summarise(n_pings=sum(n_pings), n_fishing=sum(n_fishing),
                frac=sum(n_fishing)/pmax(sum(n_pings),1),
                n_cells=mean(n_cells), .groups="drop")
    yvar <- input$trend_metric
    ggplot(d, aes(x=factor(year), y=.data[[yvar]], fill=factor(year))) +
      geom_col(width=0.65, alpha=0.88) +
      scale_fill_brewer(palette="GnBu", direction=1) +
      scale_y_continuous(
        labels=if(yvar=="frac") label_percent(scale=100)
          else if(yvar %in% c("n_pings","n_fishing"))
            label_number(scale=1/1e6,suffix="M")
          else label_comma()) +
      labs(title="Annual total", x="Year", y=NULL) +
      theme_bw(base_size=10) +
      theme(legend.position="none", panel.grid.major.x=element_blank(),
            panel.grid.minor=element_blank(),
            plot.title=element_text(face="bold",size=10))
  })
}

shinyApp(ui, server)
Data and methods
  • Source: SERNAPESCA Vessel Monitoring System (VMS), artisanal fleet
  • Period: 2019–2026 · Region: Los Lagos (lat –44.5° to –40.5°)
  • Fishing: speed 0.1–3.0 knots (Behivoke et al. 2021)
  • Grid: 0.05° × 0.05° cells. Farm distances: to nearest marine concession centroid.
  • Note: 2019–2022 from admin snapshots (~4×/day, ~1,000 vessels); 2023+ from daily artisanal reports (~175–190 vessels).
